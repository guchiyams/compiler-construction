int    low,  high, step$;      ! declarations !
int      x,     y,     z;

get (low);
get (high);
get (step$);
get (low, high, step$);
put((high + low));
put((high - low));
put((high / low));
put((high * low));

if (low => high)
    {
        return;
    } else {
        put(low);
    } endif

get(high, low);

while ( low < high ) {
    high = low;
    low = high + step$;
    put(step$);
    get(step$, low, high);
}


return low;
