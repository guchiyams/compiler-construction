function convert$ [fahr:int]
{
    return 5 * (fahr -32) / 9;
}
%%
int    low,  high, step$;      ! declarations !

get (low, high, step$);

if (low => high) {
    return;
}
endif

while (low  <  high )  
    {  put (low);
        put (convert$ (low));
        low =  low  +  step$;
    } 

return low;
