int    low,  high, step$;       ! declarations !

get (low, high, step$);
put(high);
while (low  <  high )  
    {   
        put (low);
        put (high);
        if (low == step$) {
            low =  low  +  step$;
        } else {
            high = 6;
        } endif
        low = 7;
    }

high = low + 100 + step$ + high;

if ( 500 => 10000 ) {
    return;
} endif
return high;
